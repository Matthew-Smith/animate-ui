import { Step, Steps } from 'fumadocs-ui/components/steps';

<Callout>
  **Note:** We use installation process as
  [shadcn/ui](https://ui.shadcn.com/docs/installation).
</Callout>

<Steps>
<Step>

### Initialize shadcn/ui

Run the `init` command to create a new project with shadcn/ui or setup an existing one.

```bash
npx shadcn@latest init
```

</Step>
<Step>

### Add Components

Run the `add` command to add a component to your project.

```bash
npx shadcn@latest add "https://animate-ui.com/r/writing-text"
```

</Step>
<Step>

### Import Components

Import the component to your project.

```tsx
import { WritingText } from '@/components/animate-ui/writing-text';

export default function Home() {
  return (
    <div>
      <WritingText />
    </div>
  );
}
```

</Step>
</Steps>
